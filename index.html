Quick Start Guide

Get started with data visualization in just a few lines of code

Your First Plot

# Import libraries
import
matplotlib.pyplot as plt
import
numpy as np

# Create data
x = np.linspace(0, 10, 100)
y = np.sin(x)

# Create plot
plt.figure(figsize=(8, 6))
plt.plot(x, y, label='sin(x)', color='blue')
plt.title('Sine Wave')
plt.xlabel('x')
plt.ylabel('y')
plt.legend()
plt.show()
