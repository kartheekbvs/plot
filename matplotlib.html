Introduction to Matplotlib

Matplotlib is the foundation of data visualization in Python. It provides a comprehensive set of tools for creating static, animated, and interactive visualizations. Think of it as your digital canvas for turning data into insights.

Why Matplotlib?

  • • Complete control over every visual element
  • • Publication-quality plots ready for scientific journals
  • • Extensive customization options
  • • Foundation for other libraries like Seaborn

Your First Plot

import matplotlib.pyplot as plt
import numpy as np

# Create simple data
x = [1, 2, 3, 4, 5]
y = [2, 4, 6, 8, 10]

# Create the plot
plt.figure(figsize=(8, 6))
plt.plot(x, y, 'bo-', linewidth=2, markersize=8)
plt.title('My First Plot', fontsize=16, fontweight='bold')
plt.xlabel('X Axis', fontsize=12)
plt.ylabel('Y Axis', fontsize=12)
plt.grid(True, alpha=0.3)
plt.show()

Multiple Line Plot

import matplotlib.pyplot as plt
import numpy as np

# Create data
x = np.linspace(0, 10, 100)
y1 = np.sin(x)
y2 = np.cos(x)
y3 = np.tan(x)

# Create plot
plt.figure(figsize=(10, 6))
plt.plot(x, y1, label='sin(x)', color='blue', linewidth=2)
plt.plot(x, y2, label='cos(x)', color='red', linewidth=2)
plt.plot(x, y3, label='tan(x)', color='green', linewidth=2, alpha=0.7)

plt.title('Trigonometric Functions', fontsize=16, fontweight='bold')
plt.xlabel('x')
plt.ylabel('y')
plt.legend()
plt.grid(True, alpha=0.3)
plt.ylim(-5, 5)  # Limit y-axis for better view
plt.show()

Scatter Plot with Styling

Grouped Bar Chart

import matplotlib.pyplot as plt
import numpy as np

# Data
categories = ['A', 'B', 'C', 'D', 'E']
values1 = [20, 35, 30, 25, 40]
values2 = [25, 30, 35, 20, 35]
values3 = [15, 25, 20, 30, 25]

# Create bar chart
x = np.arange(len(categories))
width = 0.25

plt.figure(figsize=(10, 6))
plt.bar(x - width, values1, width, label='Group 1', color='skyblue')
plt.bar(x, values2, width, label='Group 2', color='lightcoral')
plt.bar(x + width, values3, width, label='Group 3', color='lightgreen')

plt.title('Grouped Bar Chart Example', fontsize=16, fontweight='bold')
plt.xlabel('Categories')
plt.ylabel('Values')
plt.xticks(x, categories)
plt.legend()
plt.grid(True, alpha=0.3, axis='y')
plt.show()

Distribution Comparison

import matplotlib.pyplot as plt
import numpy as np

# Generate different distributions
np.random.seed(42)
normal_data = np.random.normal(0, 1, 1000)
uniform_data = np.random.uniform(-3, 3, 1000)
exponential_data = np.random.exponential(1, 1000)

# Create histogram
plt.figure(figsize=(12, 6))

plt.subplot(1, 3, 1)
plt.hist(normal_data, bins=30, alpha=0.7, color='blue', edgecolor='black')
plt.title('Normal Distribution')
plt.xlabel('Value')
plt.ylabel('Frequency')

plt.subplot(1, 3, 2)
plt.hist(uniform_data, bins=30, alpha=0.7, color='green', edgecolor='black')
plt.title('Uniform Distribution')
plt.xlabel('Value')
plt.ylabel('Frequency')

plt.subplot(1, 3, 3)
plt.hist(exponential_data, bins=30, alpha=0.7, color='red', edgecolor='black')
plt.title('Exponential Distribution')
plt.xlabel('Value')
plt.ylabel('Frequency')

plt.tight_layout()
plt.show()

Complete Styling Example

import matplotlib.pyplot as plt
import numpy as np

# Create data
x = np.linspace(0, 10, 100)
y1 = np.sin(x)
y2 = np.cos(x)

# Create styled plot
plt.figure(figsize=(12, 8))
plt.style.use('seaborn-v0_8')

# Plot with custom styling
plt.plot(x, y1, 
         color='#1f77b4', 
         linewidth=3, 
         linestyle='--', 
         marker='o', 
         markersize=4,
         markerfacecolor='white',
         markeredgecolor='#1f77b4',
         markeredgewidth=2,
         label='sin(x)',
         alpha=0.8)

plt.plot(x, y2, 
         color='#ff7f0e', 
         linewidth=3, 
         linestyle='-', 
         marker='s', 
         markersize=4,
         markerfacecolor='white',
         markeredgecolor='#ff7f0e',
         markeredgewidth=2,
         label='cos(x)',
         alpha=0.8)

# Customize the plot
plt.title('Styled Trigonometric Functions', 
          fontsize=20, 
          fontweight='bold', 
          fontfamily='serif',
          color='#2c3e50')

plt.xlabel('X Axis', 
           fontsize=14, 
           fontweight='bold',
           color='#34495e')
plt.ylabel('Y Axis', 
           fontsize=14, 
           fontweight='bold',
           color='#34495e')

# Customize legend
plt.legend(fontsize=12, 
           frameon=True, 
           fancybox=True, 
           shadow=True,
           loc='upper right')

# Customize grid
plt.grid(True, 
         alpha=0.3, 
         linestyle='-', 
         linewidth=1,
         color='#95a5a6')

# Customize axes
plt.gca().spines['top'].set_visible(False)
plt.gca().spines['right'].set_visible(False)
plt.gca().spines['left'].set_color('#bdc3c7')
plt.gca().spines['bottom'].set_color('#bdc3c7')

plt.tight_layout()
plt.show()

Subplots & Layouts

Subplots allow you to create multiple plots within a single figure. This is essential for comparing different datasets or showing multiple aspects of your data in one view.

Subplot Methods

plt.subplot(rows, cols, index)

Simple grid layout

plt.subplots()

Create figure and axes together

GridSpec

Complex, irregular layouts

Layout Tips

  • Use consistent scales for comparison
  • Maintain aspect ratios when possible
  • Use tight_layout() to prevent overlap
  • Consider color blindness in design

Advanced Subplot Layout

import matplotlib.pyplot as plt
import numpy as np

# Create figure with complex layout
fig = plt.figure(figsize=(15, 10))

# Create grid layout
gs = fig.add_gridspec(3, 3, hspace=0.3, wspace=0.3)

# Main plot (spans 2x2)
ax_main = fig.add_subplot(gs[0:2, 0:2])

# Side plots
ax_right = fig.add_subplot(gs[0:2, 2])
ax_bottom = fig.add_subplot(gs[2, 0:2])
ax_corner = fig.add_subplot(gs[2, 2])

# Generate data for main plot
x = np.linspace(0, 10, 100)
y1 = np.sin(x)
y2 = np.cos(x)

# Main plot
ax_main.plot(x, y1, label='sin(x)', color='blue', linewidth=2)
ax_main.plot(x, y2, label='cos(x)', color='red', linewidth=2)
ax_main.set_title('Main Plot: Trigonometric Functions', fontsize=14, fontweight='bold')
ax_main.legend()
ax_main.grid(True, alpha=0.3)

# Right plot - histogram of sin(x)
ax_right.hist(np.sin(x), bins=20, color='blue', alpha=0.7, orientation='horizontal')
ax_right.set_title('Distribution')
ax_right.set_ylabel('sin(x) values')

# Bottom plot - bar chart
categories = ['A', 'B', 'C', 'D', 'E']
values = [20, 35, 30, 25, 40]
ax_bottom.bar(categories, values, color='green', alpha=0.7)
ax_bottom.set_title('Categorical Data')
ax_bottom.set_xlabel('Categories')

# Corner plot - scatter
np.random.seed(42)
scatter_x = np.random.randn(50)
scatter_y = np.random.randn(50)
ax_corner.scatter(scatter_x, scatter_y, c='red', alpha=0.6)
ax_corner.set_title('Random Scatter')
ax_corner.set_xlabel('X')
ax_corner.set_ylabel('Y')

plt.suptitle('Complex Subplot Layout Example', fontsize=16, fontweight='bold')
plt.show()
